from nltk.wsd import lesk
from nltk.corpus import wordnet as wn

VAR TEXT —- text to read
VAR WORD —- word to define

words = file(TEXT,”r”).read().split()
print(lesk(words, ‘WORD’))
— the answer is the defined word

for ss in wn.synsets(‘WORD’):
	print(ss,ss.definition())

— one of the list is the meaning of the defined word
